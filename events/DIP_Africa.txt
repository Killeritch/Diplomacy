add_namespace = africa

# Turkish ultimatum to Egypt
country_event = {
	id = africa.1
	title = africa.1.t
	desc = africa.1.d
	picture = GFX_report_event_soviet_purge_speech
	
	is_triggered_only = yes
	
	option = { # Yield
		name = africa.1.a
		ai_chance = {
			base = 40
			modifier = {
				factor = 0				
				any_other_country = {
					is_major = yes
					NOT = { tag = TUR }
					OR = {
						has_guaranteed = EGY
						is_in_faction_with = EGY
					}
				}
			}
			modifier = {
				factor = 0.5
				OR = {
					has_government = communism
					has_government = fascism
				}
			}
		}
		TUR = {
			country_event = { days = 1 id = africa.4 }
		}
		
	}
	option = { # Resist
		name = africa.1.b
		ai_chance = {
			factor = 20 
			modifier = {
				factor = 0
				OR = {
					AND = {
						is_in_faction_with = TUR
						TUR = { is_faction_leader = yes }
					}
					AND = {
						is_in_faction = no 
						NOT = {
							any_other_country = {
								NOT = { tag = TUR }
								has_guaranteed = EGY
							} 
						}
					}
				}
			}
		}
		TUR = {
			effect_tooltip = {
				create_wargoal = {
					type = annex_everything
					target = EGY
				}
			}
			country_event = { days = 1 id = africa.5 }
		}
	}
}

###########################
# DIPLOMACY MOD Startup Events
###########################

add_namespace = startup

# AUSTRIA
country_event = {
	id = startup.1
	title = startup.1.t
	desc = startup.1.d
	picture = GFX_austria_standin
	
	trigger = {
		OR = {
			tag = ENG
			tag = FRA
			tag = GER
			tag = ITA
			tag = SOV
			tag = TUR
		}
	}
	
	is_triggered_only = yes
	
	option = {
		name = startup.1.a
		set_country_flag = aus_friendly
	}
	
	option = {
		name = startup.1.b
	}

	option = {
		name = startup.1.c
		set_country_flag = aus_hostile
		create_wargoal = {
			type = annex_everything
			target = AUS
		}
		AUS = {
			create_wargoal = {
				type = annex_everything
				target = ROOT
			}
		}
	}
}

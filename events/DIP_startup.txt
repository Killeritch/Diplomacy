# DIPLOMACY MOD Startup Events
###########################

add_namespace = startup

# AUSTRIA
country_event = {
	id = startup.1
	title = startup.1.t
	desc = startup.1.d
	picture = GFX_austria_standin
	
	trigger = {
		NOT = {
			has_country_flag = startup_aus
		}
		OR = {
			tag = ENG
			tag = FRA
			tag = GER
			tag = ITA
			tag = SOV
			tag = TUR
		}
	}
	
	mean_time_to_happen = {
		days = 7
	}
	
	immediate = {
		set_country_flag = startup_aus
	}
	
	option = {
		name = startup.1.a
		set_country_flag = aus_friendly
	}
	
	option = {
		name = startup.1.b
	}

	option = {
		name = startup.1.c
		set_country_flag = aus_hostile
		create_wargoal = {
			type = annex_everything
			target = AUS
		}
	}
	
	option = {
		name = startup.1.e
		set_country_flag = aus_hostile
		declare_war_on = {
			target = AUS
			type = annex_everything
		}
		ai_chance = {
			base = 0
		}
	}
}

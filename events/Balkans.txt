add_namespace = balkans

# Austrian ultimatum to Serbia
country_event = {
	id = balkans.1
	title = balkans.1.t
	desc = balkans.1.d
	picture = GFX_report_event_soviet_purge_speech
	
	is_triggered_only = yes
	
	option = { # Yield
		name = balkans.1.a
		ai_chance = {
			base = 40
			modifier = {
				factor = 0				
				any_other_country = {
					is_major = yes
					NOT = { tag = AUS }
					OR = {
						has_guaranteed = SER
						is_in_faction_with = SER
					}
				}
			}
			modifier = {
				factor = 0.5
				OR = {
					has_government = communism
					has_government = fascism #glorious Serbia will never yield!
				}
			}
		}
		AUS = {
			country_event = { days = 1 id = balkans.4 }
		}
		custom_effect_tooltip = GAME_OVER_TT
	}
	option = { # Resist
		name = balkans.1.b
		ai_chance = {
			factor = 20 
			modifier = {
				factor = 0
				OR = {
					AND = {
						is_in_faction_with = AUS
						AUS = { is_faction_leader = yes }
					}
					AND = {
						is_in_faction = no 
						NOT = {
							any_other_country = {
								NOT = { tag = AUS }
								has_guaranteed = SER
							} 
						}
					}
				}
			}
		}
		AUS = {
			effect_tooltip = {
				create_wargoal = {
					type = take_state_focus
					target = SER
					generator = { 107 }
				}
			}
			country_event = { days = 1 id = balkans.5 }
		}
	}
}

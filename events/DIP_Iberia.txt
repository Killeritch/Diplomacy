# British/Italian ultimatum to Spain (Gibraltar)
country_event = {
	id = iberia.1
	title = iberia.1.t
	desc = iberia.1.d
	picture = GFX_report_event_soviet_purge_speech
	
	is_triggered_only = yes
	
	option = { # Yield
		name = iberia.1.a
		ai_chance = {
			base = 40
			modifier = {
				factor = 0				
				any_other_country = {
					is_major = yes
					NOT = { tag = FROM }
					OR = {
						has_guaranteed = SPR
						is_in_faction_with = SPR
					}
				}
			}
			modifier = {
				factor = 0.5
				OR = {
					has_government = communism
					has_government = fascism
				}
			}
		}
		FROM = {
			country_event = { days = 1 id = iberia.4 }
		}
		
	}
	option = { # Resist
		name = iberia.1.b
		ai_chance = {
			factor = 20 
			modifier = {
				factor = 0
				OR = {
					AND = {
						is_in_faction_with = FROM
						FROM = { is_faction_leader = yes }
					}
					AND = {
						is_in_faction = no 
						NOT = {
							any_other_country = {
								NOT = { tag = FROM }
								has_guaranteed = SPR
							} 
						}
					}
				}
			}
		}
		FROM = {
			effect_tooltip = {
				create_wargoal = {
					type = take_state_focus
					target = SPR
					generator = { 118 }
				}
			}
			country_event = { days = 1 id = iberia.5 }
		}
	}
}

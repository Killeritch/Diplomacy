# AUSTRIA
country_event = {
	id = alliance.1
	title = alliance.1.t
	desc = alliance.1.d
	picture = GFX_austria_standin
	
	trigger = {
		has_country_flag = aus_friendly
	}
	
	mean_time_to_happen = {
		days = 30
	}
	
	immediate = {
		clr_country_flag = aus_friendly
	}
	
	option = { # no we're good
		name = alliance.1.a		
	}
	
	option = { #offer defensive pact
		name = alliance.1.b
		AUS = {
			country_event = {
				id = alliance.10
				days = 1
			}
		}
	}
	
	option = { #alliance against Germany
		name = alliance.1.c
		trigger = {
			OR = {
				tag = ENG
				tag = FRA
				tag = SOV
			}
			NOT = {
				has_country_flag = ger_friendly
			}
		}
		AUS = {
			country_event = {
				id = plot.4
				days = 1
			}
		}
	}

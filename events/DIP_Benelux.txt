	id = benelux.1
	title = benelux.1.t
	desc = benelux.1.d
	picture = GFX_uk_standin
	
	is_triggered_only = yes
	
	option = { # Yield
		name = benelux.1.a
		ai_chance = {
			base = 40
			modifier = {
				factor = 0				
				any_other_country = {
					is_major = yes
					NOT = { tag = ENG }
					OR = {
						has_guaranteed = HOL
						is_in_faction_with = HOL
					}
				}
			}
			modifier = {
				factor = 0.5
				OR = {
					has_government = communism
					has_government = fascism
				}
			}
		}
		ENG = {
			country_event = { days = 1 id = benelux.4 }
		}
		
	}
	option = { # Resist
		name = benelux.1.b
		ai_chance = {
			factor = 20 
			modifier = {
				factor = 0
				OR = {
					AND = {
						is_in_faction_with = ENG
						ENG = { is_faction_leader = yes }
					}
					AND = {
						is_in_faction = no 
						NOT = {
							any_other_country = {
								NOT = { tag = ENG }
								has_guaranteed = HOL
							} 
						}
					}
				}
			}
		}
		ENG = {
			effect_tooltip = {
				create_wargoal = {
					type = take_state_focus
					target = HOL
					generator = { 7 }
				}
			}
			country_event = { days = 1 id = benelux.5 }
		}
	}
}

country_event = {
	id = benelux.11
	title = benelux.11.t
	desc = benelux.11.d
	picture = GFX_uk_standin
	
	trigger = {
		OR = {
			tag = FRA
			tag = GER
		}
		NOT = { has_country_flag = benelux_pro_ENG }
		NOT = { has_country_flag = benelux_anti_ENG }
		ENG = {
			NOT = { has_completed_focus = DIP_ENG_benelux_1 }
			focus_progress = {
				focus = DIP_ENG_benelux_1
				progress > 0.1
			}
		}
	}

	mean_time_to_happen = {
		days = 2
	}

	option = { # let it slide
		name = benelux.11.a
		set_country_flag = benelux_pro_ENG
		ai_chance = {
			base = 50
			modifier = {
				factor = 0
				has_war_with = ENG
			}
		}
	}
	
	option = { # guarantee Netherlands
		name = benelux.11.b
		set_country_flag = benelux_anti_ENG
		if = {
			limit = {
				NOT = { has_guaranteed = HOL }
			}
			give_guarantee = HOL
		}
		ai_chance = {
			base = 50
			modifier = {
				factor = 0.1
				surrender_progress > 0.1
			}
			modifier = {
				factor = 0
				is_in_faction_with = ENG
			}
		}
	}
}

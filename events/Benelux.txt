add_namespace = benelux

# UK ultimatum to Netherlands
country_event = {
	id = benelux.1
	title = benelux.1.t
	desc = benelux.1.d
	picture = GFX_report_event_soviet_purge_speech
	
	is_triggered_only = yes
	
	option = { # Yield
		name = benelux.1.a
		ai_chance = {
			base = 40
			modifier = {
				factor = 0				
				any_other_country = {
					is_major = yes
					NOT = { tag = ENG }
					OR = {
						has_guaranteed = HOL
						is_in_faction_with = HOL
					}
				}
			}
			modifier = {
				factor = 0.5
				OR = {
					has_government = communism
					has_government = fascism #glorious Netherlands will never yield!
				}
			}
		}
		ENG = {
			country_event = { days = 1 id = benelux.4 }
		}
		custom_effect_tooltip = GAME_OVER_TT
	}
	option = { # Resist
		name = benelux.1.b
		ai_chance = {
			factor = 20 
			modifier = {
				factor = 0
				OR = {
					AND = {
						is_in_faction_with = ENG
						ENG = { is_faction_leader = yes }
					}
					AND = {
						is_in_faction = no 
						NOT = {
							any_other_country = {
								NOT = { tag = ENG }
								has_guaranteed = HOL
							} 
						}
					}
				}
			}
		}
		ENG = {
			effect_tooltip = {
				create_wargoal = {
					type = take_state_focus
					target = HOL
					generator = { 7 }
				}
			}
			country_event = { days = 1 id = benelux.5 }
		}
	}
}
